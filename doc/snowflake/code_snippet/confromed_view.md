        <COLUMN 1 NAME>, --view columns list ✏️
        <COLUMN 2 NAME>,
        ...
        <COLUMN N NAME>,
        EXTRACTEDON, ✏️
        HASH_VALUE

/*----------------------------------------------------------------------------------------------
 This view is for loading data into <SCHEMA NAME>.<TABLE NAME>
 -----------------------------------------------------------------------------------------------
 Revision History
 -----------------------------------------------------------------------------------------------
 Date            Developer                  Comments
 -----------------------------------------------------------------------------------------------
 YYYY-MM-DD      <YOUR NAME>                Initial Revision
 YYYY-MM-DD      <YOUR NAME>                <Comments for new version>
 -----------------------------------------------------------------------------------------------
 */
SELECT
    <COLUMN 1 NAME>,
    <COLUMN 2 NAME>,
    ...
    <COLUMN N NAME>,
    EXTRACTEDON, ✏️
    HASH(<All Columns List except FRMW_EXTRACTED_ON>) AS HASH_VALUE ✏️
FROM
    CONFORMED.<TABLE NAME>;
